19      public CallVisitVertex( int fieldVar,

21  			    short position ) {
22  	this.position = position;
23  	InstructionList il = new InstructionList();
24  	InstructionList end = new InstructionList();
25  	end.append( new LabeledNOP("END VISIT(" + position + ")") );

27  	il.append( new ALOAD( fieldVar ));
28  	il.append( new IFNULL( end.getStart() ));
29  	il.append( new ALOAD( fieldVar ));

32  	il.append( end );
33  
34  	setInstructionList( il );
35  
36  	//      ALOAD
37  	//        |
38          //      IFNULL
39  	//      T/  \F

42  	//      |   SIPUSH
43  	//      |     |
44  	//      |   INVOKEVIRTUAL
45  	//       \   /
46  	//        NOP
47  	//
48      }
